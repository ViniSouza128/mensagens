Implementação

let audioCtx;
function tone(freq, dur=0.08, type='sine', vol=0.05) {
  if (!soundsOn) return;
  const ctx = audioCtx ||= new (AudioContext||webkitAudioContext)();
  const o = ctx.createOscillator(), g = ctx.createGain();
  o.type = type;
  o.frequency.setValueAtTime(freq, ctx.currentTime);
  g.gain.setValueAtTime(vol, ctx.currentTime);
  g.gain.exponentialRampToValueAtTime(0.0001, ctx.currentTime + dur);
  o.connect(g); g.connect(ctx.destination);
  o.start(); o.stop(ctx.currentTime + dur);
}

Regras

Sempre opt-in/out: usuário pode desligar nos settings.
Volume baixo (3-6%): som não pode interromper o foco.
Curto: nada acima de 200ms.
Não toque ao abrir o app. Web Audio precisa de gesto do usuário.
